00023 /*----------------------------------------------------------------------------------------------------------------------
00024 |       Class for holding discrete states in a matrix. Note that there is no way to access the variables of this class 
00025 |       since they are all private and there are no public access functions. This class is designed to be manipulated by 
00026 |       the class NxsDiscreteMatrix, which is the only class that has been designated a friend of NxsDiscreteDatum. 
00027 |       The variable `states' is NULL if there is missing data, and non-NULL for any other state. If `states' is non-NULL,
00028 |       the first cell is used to store the number of states. This will be 0 if the state is the gap state, 1 if the state
00029 |       is unambiguous and nonpolymorphic (and not the gap state of course), and 2 or higher if there is either 
00030 |       polymorphism or uncertainty. If polymorphism or uncertainty apply, it becomes necessary to store information about 
00031 |       which of these two situations holds. Thus, the last cell in the array is set to either 1 (polymorphism) or 0 
00032 |       (uncertainty). While a little complicated, this scheme has the following benefits:
00033 |~
00034 |       o if the state is missing, the only memory allocated is for a pointer (`states')
00035 |       o if the state is unambiguous and not polymorphic, no storage is used for keeping track of whether polymorphism or 
00036 |         uncertainty holds
00037 |       o it allows for a virtually unlimited number of states, which is important if it is to be general enough to store 
00038 |         microsatellite data for a NxsAllelesBlock object, for example.
00039 |~
00040 |       Supposing the gap symbol is '-', the missing data symbol is '?', and the symbols list is "ACGT", the following 
00041 |       table shows the status of the states variable under several different possible data matrix entries:
00042 |>
00043 |       Matrix entry        states array
00044 |       --------------------------------
00045 |            ?              NULL
00046 |            -              [0]
00047 |            G              [1][2]
00048 |       (AG) polymorphic    [2][0][2][1]
00049 |       {AG} ambiguous      [2][0][2][0]
00050 |       --------------------------------
00051 |>      
00052 */
00053 class NxsDiscreteDatum
00054         {
00055         friend class NxsDiscreteMatrix;

00057         public:
00058 
00059                                         NxsDiscreteDatum();
00060                 virtual         ~NxsDiscreteDatum();
00061 
00062                 void            CopyFrom(const NxsDiscreteDatum &other);
00063 
00064         private:
00065 
00066                 unsigned        *states;        /* holds information about state for a single taxon-character combination */
00067         };
